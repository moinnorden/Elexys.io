ESP32 / ESP8266 (Arduino)
#include <HTTPClient.h>
#include <ArduinoJson.h>

const char* TOKEN = "elx_YOUR_TOKEN";
const char* URL =
  "https://elexys.io/api/data";

void sendData(float temp, float hum) {
  HTTPClient http;
  http.begin(URL);
  http.addHeader("Authorization",
    String("Bearer ") + TOKEN);
  http.addHeader("Content-Type",
    "application/json");

  StaticJsonDocument<200> doc;
  doc["temperature"] = temp;
  doc["humidity"] = hum;
  doc["board"] = "ESP32";

  String body;
  serializeJson(doc, body);
  http.POST(body);
  http.end();
}
Raspberry Pi (Python)
import requests, json

TOKEN = "elx_YOUR_TOKEN"
URL = "https://elexys.io/api/data"

def send_data(temp, hum):
    headers = {
        "Authorization": f"Bearer {TOKEN}",
        "Content-Type": "application/json"
    }
    payload = {
        "temperature": temp,
        "humidity": hum,
        "board": "RaspberryPi"
    }
    r = requests.post(URL,
        headers=headers,
        json=payload)
    print(r.status_code)
